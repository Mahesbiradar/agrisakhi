--- src/utils/location.js ---
export async function detectLocation():
  Returns Promise<{lat, lng, village, district}>
  
  Steps:
  1. navigator.geolocation.getCurrentPosition() wrapped in Promise
  2. On success: call Nominatim reverse geocode:
     fetch(`https://nominatim.openstreetmap.org/reverse?lat=${lat}&format=json&lon=${lng}&accept-language=en`)
  3. Parse response.address:
     village = address.village || address.suburb || address.town || address.hamlet || ''
     district = address.county || address.state_district || ''
  4. Return {lat, lng, village, district}
  5. On geolocation error or fetch error: throw Error('location_failed')

export async function uploadImage(file):
  FormData with file + upload_preset + folder='agrisakhi/images'
  POST to CLOUDINARY_URL
  Returns secure_url string

export async function uploadAudio(blob):
  FormData with blob as file named 'audio.webm' + upload_preset + resource_type='video' + folder='agrisakhi/audio'
  POST to CLOUDINARY_URL
  Returns secure_url string

--- src/utils/audio.js ---
export class AudioRecorder:
  properties: mediaRecorder, chunks, stream
  
  async start():
    Request microphone: navigator.mediaDevices.getUserMedia({audio: true})
    Create MediaRecorder(stream, {mimeType: 'audio/webm'})
    mediaRecorder.ondataavailable: push to chunks
    mediaRecorder.start()
  
  stop(): Returns Promise<Blob>
    mediaRecorder.stop()
    On onstop: resolve(new Blob(chunks, {type:'audio/webm'}))
    Stop all stream tracks
  
  isSupported(): return !!navigator.mediaDevices?.getUserMedia
