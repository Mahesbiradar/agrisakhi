--- FRONTEND FIX: src/lib/api.js ---

Update the response interceptor error handling:

response interceptor on error:
  const status = error.response?.status
  const data = error.response?.data

  if (!error.response) {
    // Network error — CORS or server down
    error.userMessage = 'Cannot connect to server. Make sure backend is running.'
    return Promise.reject(error)
  }

  if (status === 400) {
    // Extract first error message from Django's response format
    if (data?.phone) error.userMessage = 'Phone number already registered.'
    else if (data?.non_field_errors) error.userMessage = data.non_field_errors[0]
    else if (data?.detail) error.userMessage = data.detail
    else {
      // Get first value from error object
      const firstKey = Object.keys(data || {})[0]
      error.userMessage = firstKey ? `${firstKey}: ${data[firstKey][0]}` : 'Invalid data submitted.'
    }
  } else if (status === 401) {
    error.userMessage = 'Session expired. Please login again.'
  } else if (status === 403) {
    error.userMessage = 'You do not have permission for this action.'
  } else if (status === 404) {
    error.userMessage = 'Resource not found.'
  } else {
    error.userMessage = 'Something went wrong. Please try again.'
  }

  // Handle 401 refresh logic (keep existing refresh token logic here)
  if (status === 401) {
    // ... existing refresh token code
  }

  return Promise.reject(error)
