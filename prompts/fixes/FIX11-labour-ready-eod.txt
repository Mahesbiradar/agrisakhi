class ReadyForWorkView(APIView):
    permission_classes = [IsAuthenticated]
    def post(self, request):
        from datetime import date
        is_ready = request.data.get('is_ready', True)
        request.user.is_ready_for_work = is_ready
        if is_ready:
            request.user.ready_until = date.today()
        else:
            request.user.ready_until = None
        request.user.save(update_fields=['is_ready_for_work', 'ready_until'])
        return Response({'is_ready': is_ready})

EOD auto-reset logic (check on every dashboard load):
const checkReadyReset = () => {
  const lastDate = localStorage.getItem('agrisakhi_ready_date')
  const today = new Date().toDateString()
  if (lastDate !== today) {
    // New day — reset ready status
    localStorage.setItem('agrisakhi_ready_date', today)
    setIsReady(false)
    usersAPI.setReadyForWork(false).catch(() => {})
  }
}
